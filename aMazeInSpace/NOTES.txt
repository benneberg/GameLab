Try additional sound effects for game interactions (optional)
You could also add subtle rolling sounds for the ball or "ping" sounds when passing certain points:

// Add to your update function for rolling sounds
let lastPlayedRollSound = 0;
function update() {
  if (!running || gameOver) return;
  
  // Add rolling sound if ball is moving fast enough
  const speed = Math.sqrt(ball.vx * ball.vx + ball.vy * ball.vy);
  if (speed > 1 && audioEnabled) {
    const now = Date.now();
    if (now - lastPlayedRollSound > 300) { // Play every 300ms at most
      // Create a "rolling" sound - very subtle
      const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      const oscillator = audioCtx.createOscillator();
      const gainNode = audioCtx.createGain();
      
      oscillator.type = 'sine';
      oscillator.frequency.value = 100 + speed * 20; // Higher pitch with higher speed
      oscillator.connect(gainNode);
      
      gainNode.gain.value = 0.03; // Very quiet
      gainNode.connect(audioCtx.destination);
      
      oscillator.start();
      setTimeout(() => {
        oscillator.stop();
      }, 50); // Very short duration
      
      lastPlayedRollSound = now;
    }
  }
  
  // ... rest of update function
}
